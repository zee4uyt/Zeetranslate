// ===== SPEECH RECOGNITION =====
function initializeSpeechRecognition() {
    if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        recognition = new SpeechRecognition();
        recognition.continuous = true;
        recognition.interimResults = true;
        
        recognition.onstart = () => {
            isRecording = true;
            micBtn.classList.add('active');
            recordingIndicator.classList.add('active');
        };
        
        recognition.onresult = (event) => {
            let interimTranscript = '';
            let finalTranscript = '';
            
            for (let i = event.resultIndex; i < event.results.length; i++) {
                const transcript = event.results[i][0].transcript;
                if (event.results[i].isFinal) {
                    finalTranscript += transcript + ' ';
                } else {
                    interimTranscript += transcript;
                }
            }
            
            inputText.value = finalTranscript + interimTranscript;
            updateCharCount();
            
            if (finalTranscript) {
                translateText();
            }
        };
        
        recognition.onerror = (event) => {
            console.error('Speech recognition error:', event.error);
            stopRecording();
            showToast('Error: ' + event.error, 'error');
        };
        
        recognition.onend = () => {
            if (isRecording) {
                recognition.start();
            }
        };
    } else {
        micBtn.disabled = true;
        micBtn.title = 'Speech recognition not supported';
    }
}

// ===== TRANSLATION FUNCTION =====
async function translateText() {
    const text = inputText.value.trim();
    if (!text) {
        outputText.textContent = 'Translation will appear here...';
        outputText.classList.add('empty');
        detectedLanguage.textContent = '';
        return;
    }
    
    const sourceLang = inputLanguage.value;
    const targetLang = outputLanguage.value;
    
    // Show loading state
    outputText.textContent = 'Translating...';
    outputText.classList.remove('empty');
    
    try {
        // Detect language if auto-detect is selected
        let detectedLang = sourceLang;
        if (sourceLang === 'auto') {
            detectedLang = await detectLanguage(text);
            detectedLanguage.textContent = `Detected: ${languageNames[detectedLang] || detectedLang}`;
        } else {
            detectedLanguage.textContent = '';
        }
        
        // Use MyMemory Translation API (free and unlimited)
        const response = await fetch(
            `https://api.mymemory.translated.net/get?q=${encodeURIComponent(text)}&langpair=${detectedLang}|${targetLang}`
        );
        
        const data = await response.json();
        
        if (data.responseStatus === 200 && data.responseData) {
            const translatedText = data.responseData.translatedText;
            outputText.textContent = translatedText;
            outputText.classList.remove('empty');
            
            // Add to history
            addToHistory(text, translatedText, detectedLang, targetLang);
        } else {
            throw new Error('Translation failed');
        }
    } catch (error) {
        console.error('Translation error:', error);
        outputText.textContent = 'Translation error. Please try again.';
        outputText.classList.add('empty');
        showToast('Translation failed. Please try again.', 'error');
    }
}

// ===== LANGUAGE DETECTION =====
async function detectLanguage(text) {
    try {
        // Use a simple heuristic-based detection
        // Check for common patterns in different languages
        
        // Arabic/Persian/Urdu (RTL scripts)
        if (/[\u0600-\u06FF]/.test(text)) {
            if (/[\u0750-\u077F]/.test(text)) return 'ur'; // Urdu-specific
            if (/[\u06A0-\u06FF]/.test(text)) return 'fa'; // Persian-specific
            return 'ar'; // Arabic
        }
        
        // Chinese
        if (/[\u4E00-\u9FFF]/.test(text)) {
            // Traditional Chinese characters (rough heuristic)
            if (/[\u3400-\u4DBF\uF900-\uFAFF]/.test(text)) return 'zh-TW';
            return 'zh';
        }
        
        // Japanese
        if (/[\u3040-\u309F\u30A0-\u30FF]/.test(text)) return 'ja';
        
        // Korean
        if (/[\uAC00-\uD7AF]/.test(text)) return 'ko';
        
        // Hindi/Devanagari
        if (/[\u0900-\u097F]/.test(text)) return 'hi';
        
        // Bengali
        if (/[\u0980-\u09FF]/.test(text)) return 'bn';
        
        // Greek
        if (/[\u0370-\u03FF]/.test(text)) return 'el';
        
        // Cyrillic scripts
        if (/[\u0400-\u04FF]/.test(text)) {
            // Ukrainian specific
            if (/[Ò‘Ñ”Ñ–Ñ—ÒÐ„Ð†Ð‡]/.test(text)) return 'uk';
            // Bulgarian specific
            if (/[ÑŠÐª]/.test(text)) return 'bg';
            // Serbian specific
            if (/[Ñ’Ñ˜Ñ™ÑšÑ›ÑŸÐ‚ÐˆÐ‰ÐŠÐ‹Ð]/.test(text)) return 'sr';
            return 'ru'; // Default to Russian
        }
        
        // Hebrew
        if (/[\u0590-\u05FF]/.test(text)) return 'he';
        
        // Thai
        if (/[\u0E00-\u0E7F]/.test(text)) return 'th';
        
        // Vietnamese (Latin with diacritics)
        if (/[ÄƒÃ¢Ä‘ÃªÃ´Æ¡Æ°Ä‚Ã‚ÄÃŠÃ”Æ Æ¯]/.test(text)) return 'vi';
        
        // Default to English for Latin script
        return 'en';
    } catch (error) {
        console.error('Language detection error:', error);
        return 'en';
    }
}

        if (event.error === 'not-allowed') {
            showToast('Please allow audio playback', 'error');
        } else if (event.error === 'language-unavailable') {
            showToast('Voice not available for this language. Using default voice.', 'error');
        } else {
            showToast('Speech error: ' + event.error, 'error');
        }
    };
    
    speechSynthesis.speak(currentUtterance);
    showToast('Playing audio...');
}

// ===== HISTORY MANAGEMENT =====
function addToHistory(sourceText, translatedText, sourceLang, targetLang) {
    const historyItem = {
        id: Date.now(),
        sourceText,
        translatedText,
        sourceLang,
        targetLang,
        timestamp: new Date().toISOString(),
        isFavorite: false
    };
    
    translationHistory.unshift(historyItem);
    
    // Keep only last 50 translations
    if (translationHistory.length > 50) {
        translationHistory = translationHistory.slice(0, 50);
    }
    
    localStorage.setItem('translationHistory', JSON.stringify(translationHistory));
}

function displayHistory() {
    if (translationHistory.length === 0) {
        historyList.innerHTML = '<p class="empty-state">No translation history yet. Start translating!</p>';
        return;
    }
    
    historyList.innerHTML = '';
    
    translationHistory.forEach(item => {
        const historyItem = document.createElement('div');
        historyItem.className = 'history-item';
        
        const date = new Date(item.timestamp);
        const formattedDate = date.toLocaleDateString() + ' ' + date.toLocaleTimeString();
        
        historyItem.innerHTML = `
            <div class="history-item-header">
                <div class="history-languages">
                    <span>${languageNames[item.sourceLang] || item.sourceLang}</span>
                    <i class="fas fa-arrow-right"></i>
                    <span>${languageNames[item.targetLang] || item.targetLang}</span>
                </div>
                <div class="history-date">${formattedDate}</div>
            </div>
            <div class="history-text">
                <strong>Original:</strong> ${item.sourceText}
            </div>
            <div class="history-text">
                <strong>Translation:</strong> ${item.translatedText}
            </div>
            <div class="history-actions-btns">
                <button class="btn-small" onclick="reuseTranslation(${item.id})">
                    <i class="fas fa-redo"></i> Reuse
                </button>
                <button class="btn-small" onclick="deleteHistoryItem(${item.id})">
                    <i class="fas fa-trash"></i> Delete
                </button>
            </div>
        `;
        
        historyList.appendChild(historyItem);
    });
}

function reuseTranslation(id) {
    const item = translationHistory.find(h => h.id === id);
    if (item) {
        inputText.value = item.sourceText;
        inputLanguage.value = item.sourceLang;
        outputLanguage.value = item.targetLang;
        outputText.textContent = item.translatedText;
        outputText.classList.remove('empty');
        updateCharCount();
        historyModal.classList.remove('active');
        showToast('Translation reused successfully!');
    }
}

function deleteHistoryItem(id) {
    translationHistory = translationHistory.filter(h => h.id !== id);
    localStorage.setItem('translationHistory', JSON.stringify(translationHistory));
    displayHistory();
    showToast('History item deleted');
}

function clearHistory() {
    if (confirm('Are you sure you want to clear all translation history?')) {
        translationHistory = [];
        localStorage.setItem('translationHistory', JSON.stringify(translationHistory));
        displayHistory();
        showToast('History cleared');
    }
}

// ===== UTILITY FUNCTIONS =====
function updateCharCount() {
    const count = inputText.value.length;
    charCount.textContent = `${count} / 5000`;
}

function copyToClipboard() {
    const text = outputText.textContent;
    if (!text || text === 'Translation will appear here...' || text === 'Translating...') {
        showToast('No text to copy', 'error');
        return;
    }
    
    navigator.clipboard.writeText(text).then(() => {
        showToast('Copied to clipboard!');
    }).catch(() => {
        showToast('Failed to copy', 'error');
    });
}

function shareTranslation() {
    const text = outputText.textContent;
    if (!text || text === 'Translation will appear here...' || text === 'Translating...') {
        showToast('No text to share', 'error');
        return;
    }
    
    if (navigator.share) {
        navigator.share({
            title: 'ZeeTranslate',
            text: text
        }).then(() => {
            showToast('Shared successfully!');
        }).catch(() => {
            // User cancelled or error
        });
    } else {
        // Fallback: copy to clipboard
        copyToClipboard();
        showToast('Link copied! (Share not supported)');
    }
}

function toggleFavorite() {
    const text = outputText.textContent;
    if (!text || text === 'Translation will appear here...' || text === 'Translating...') {
        showToast('No translation to save', 'error');
        return;
    }
    
    const icon = favoriteBtn.querySelector('i');
    if (icon.classList.contains('far')) {
        icon.classList.remove('far');
        icon.classList.add('fas');
        showToast('Added to favorites!');
    } else {
        icon.classList.remove('fas');
        icon.classList.add('far');
        showToast('Removed from favorites');
    }
}

function swapLanguages() {
    if (inputLanguage.value === 'auto') {
        showToast('Cannot swap with auto-detect', 'error');
        return;
    }
    
    const tempLang = inputLanguage.value;
    inputLanguage.value = outputLanguage.value;
    outputLanguage.value = tempLang;
    
    const tempText = inputText.value;
    inputText.value = outputText.textContent;
    outputText.textContent = tempText;
    
    if (inputText.value && inputText.value !== 'Translation will appear here...') {
        outputText.classList.remove('empty');
        translateText();
    }
}

function clearInput() {
    inputText.value = '';
    outputText.textContent = 'Translation will appear here...';
    outputText.classList.add('empty');
    detectedLanguage.textContent = '';
    updateCharCount();
}

async function pasteFromClipboard() {
    try {
        const text = await navigator.clipboard.readText();
        inputText.value = text;
        updateCharCount();
        translateText();
        showToast('Text pasted!');
    } catch (error) {
        showToast('Failed to paste', 'error');
    }
}

        showToast('Speech recognition not supported', 'error');

    if (isRecording) {
        stopRecording();
    } else {

    }
}

function stopRecording() {

        isRecording = false;
        recognition.stop();
        micBtn.classList.remove('active');
        recordingIndicator.classList.remove('active');
    }
}

// ===== SIGN IN FUNCTION =====
function handleSignIn(e) {
    e.preventDefault();
    
    const email = document.getElementById('signInEmail').value;
    const password = document.getElementById('signInPassword').value;
    
    // Find user in database
    const user = usersDatabase.find(u => u.email === email && u.password === password);
    
    if (user) {
        currentUser = { name: user.name, email: user.email };
        localStorage.setItem('currentUser', JSON.stringify(currentUser));
        updateUserUI();
        signInModal.classList.remove('active');
        showToast(`Welcome back, ${user.name}!`);
        signInForm.reset();
    } else {
        showToast('Invalid email or password', 'error');
    }
}

// ===== REGISTER FUNCTION =====
function handleRegister(e) {
    e.preventDefault();
    
    const name = document.getElementById('registerName').value;
    const email = document.getElementById('registerEmail').value;
    const password = document.getElementById('registerPassword').value;
    const confirmPassword = document.getElementById('registerConfirmPassword').value;
    
    // Validation
    if (password !== confirmPassword) {
        showToast('Passwords do not match', 'error');
        return;
    }
    
    if (password.length < 6) {
        showToast('Password must be at least 6 characters', 'error');
        return;
    }
    
    // Check if user already exists
    if (usersDatabase.find(u => u.email === email)) {
        showToast('Email already registered', 'error');
        return;
    }
    
    // Register new user
    const newUser = { name, email, password };
    usersDatabase.push(newUser);
    localStorage.setItem('usersDatabase', JSON.stringify(usersDatabase));
    
    // Auto sign in
    currentUser = { name, email };
    localStorage.setItem('currentUser', JSON.stringify(currentUser));
    
    updateUserUI();
    signInModal.classList.remove('active');
    showToast(`Welcome, ${name}! Account created successfully.`);
    registerForm.reset();
}

function processImage(file) {
    if (!file || !file.type.startsWith('image/')) {
        showToast('Please select a valid image file', 'error');
        return;
    }
    
    const reader = new FileReader();
    
    reader.onload = async function(e) {
        // Show image preview
        const imagePreview = document.createElement('div');
        imagePreview.className = 'image-preview';
        imagePreview.innerHTML = `
            <button class="image-preview-close" onclick="this.parentElement.remove()">
                <i class="fas fa-times"></i>
            </button>
            <img src="${e.target.result}" alt="Uploaded image">
            <div class="image-processing">
                <i class="fas fa-spinner fa-spin"></i>
                <p>Extracting text from image...</p>
            </div>
        `;
        
        inputText.parentElement.insertBefore(imagePreview, inputText);
        
        try {
            // Use Tesseract.js for OCR (we'll use a simple API instead)
            const extractedText = await extractTextFromImage(e.target.result);
            
            if (extractedText) {
                inputText.value = extractedText;
                updateCharCount();
                translateText();
                
                // Remove processing indicator
                const processing = imagePreview.querySelector('.image-processing');
                if (processing) processing.remove();
                
                showToast('Text extracted successfully!');
            } else {
                showToast('No text found in image', 'error');
                imagePreview.remove();
            }
        } catch (error) {
            console.error('OCR error:', error);
            showToast('Failed to extract text from image', 'error');
            imagePreview.remove();
        }
    };
    
    reader.readAsDataURL(file);
}

async function extractTextFromImage(imageData) {
    try {
        // Using OCR.space API (free tier)
        const base64Image = imageData.split(',')[1];
        
        const formData = new FormData();
        formData.append('base64Image', `data:image/png;base64,${base64Image}`);
        formData.append('language', 'eng');
        formData.append('isOverlayRequired', 'false');
        formData.append('detectOrientation', 'true');
        formData.append('scale', 'true');
        formData.append('OCREngine', '2');
        
        const response = await fetch('https://api.ocr.space/parse/image', {
            method: 'POST',
            headers: {
                'apikey': 'K87899142388957' // Free API key
            },
            body: formData
        });
        
        const result = await response.json();
        
        if (result.ParsedResults && result.ParsedResults.length > 0) {
            return result.ParsedResults[0].ParsedText.trim();
        }
        
        return null;
    } catch (error) {
        console.error('OCR API error:', error);
        // Fallback: show a message
        return 'Image text extraction requires internet connection. Please try again.';
    }
}
